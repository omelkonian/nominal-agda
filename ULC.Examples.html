-- ** α-equivalence

_ = (` a) ≡α (` a) ∋ ν≡

_ : (ƛ a ⇒ ` a) ≡α (ƛ b ⇒ ` b)
_ = ζ≡ (-, qed)
  where qed : ∀ y → y L.Mem.∉ [] → swap y a (` a) ≡α swap y b (` b)
        qed y _ rewrite swapʳ y a | swapʳ y b = ν≡

h : (ƛ a ⇒ ` c · ` a) ≡α (ƛ b ⇒ ` c · ` b)
h = ζ≡ (-, qed)
  where qed : ∀ y → y L.Mem.∉ [ c ] → swap y a (` c · ` a) ≡α swap y b (` c · ` b)
        qed y y∉ rewrite swapʳ y a | swapʳ y b
               | swap-noop y a c (λ where (here refl) → y∉ auto; (there (here ())))
               | swap-noop y b c (λ where (here refl) → y∉ auto; (there (here ())))
               = ξ≡ ν≡ ν≡

_ : (ƛ a ⇒ ` c) ≡α (ƛ b ⇒ ` c)
_ = ζ≡ (-, qed)
  where
    qed : ∀ y → y L.Mem.∉ [ c ] → swap y a (` c) ≡α swap y b (` c)
    qed y y∉
     rewrite swap-noop y a c (λ where (here refl) → y∉ auto; (there (here ())))
           | swap-noop y b c (λ where (here refl) → y∉ auto; (there (here ())))
           = ν≡

-- ** finite support

ex : Term
ex = ƛ a ⇒ ` a · ` a

suppEx = Atoms ∋ []
suppEx⁺ = Atoms ∋ [ a ]

finEx : FinSupp ex
finEx = -, go
  where
    go : ∀ 𝕒 𝕓 → 𝕒 ∉ suppEx⁺ → 𝕓 ∉ suppEx⁺ → swap 𝕓 𝕒 ex ≡α ex
    go 𝕒 𝕓 𝕒∉ 𝕓∉
      rewrite swap-noop 𝕓 𝕒 a (λ where ♯0 → 𝕓∉ auto; ♯1 → 𝕒∉ auto)
            = ≡α-refl _

_ = finEx .proj₁ ≡ suppEx⁺
  ∋ refl

finEx′ : FinSupp ex
finEx′ = ∀fin ex

_ = finEx′ .proj₁ ≢ suppEx
  ∋ λ ()

-- ** substitution

_ = (` a) [ a / ` b ] ≡ ` b
  ∋ refl

_ = (` a) [ a / ` b · ` b ] ≡ ` b · ` b
  ∋ refl

_ = (` a · ` a) [ a / ` b ] ≡ ` b · ` b
  ∋ refl

_ = (` a · ` a) [ a / ` b · ` b ] ≡ (` b · ` b) · (` b · ` b)
  ∋ refl

a' = $ 22 -- a + b

_ = (ƛ a ⇒ ` a) [ a / ` b ] ≡ (ƛ a' ⇒ ` a')
  ∋ refl

_ = (` a · (ƛ a ⇒ ` a)) [ a / ` b ] ≡ ` b · (ƛ a' ⇒ ` a')
  ∋ refl

b' = $ 24 -- b + c

_ = (ƛ b ⇒ ` b) [ b / ` c ] ≡ (ƛ b' ⇒ ` b')
  ∋ refl

c' = $ 34 -- a + b + c

_ = (` a · (ƛ c ⇒ ` c · ` a)) [ a / ` b ] ≡ (` b · (ƛ c' ⇒ ` c' · ` b))
  ∋ refl

_ = (` a · (ƛ c ⇒ ` c · ` a)) [ a / ` c' ] ≢ (` c' · (ƛ c' ⇒ ` c' · ` c'))
  ∋ λ ()

  ∋ refl
